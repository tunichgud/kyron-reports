Gitea #37416 — Missing colors in Note and Warning markdown highlighting

Reproduction report  ·  github.com/go-gitea/gitea/issues/37416  ·  BUG CONFIRMED — v1.26.1

Bug Summary

Affected: Gitea v1.26.0 and v1.26.1 (regression from PR #37019)

Symptom: GitHub-style markdown [!TIP] and [!WARNING] alert blocks render in monochrome (body text color) instead of their expected distinct colors (green / yellow-orange).

Root cause: PR #37019 re-scoped --color-success-text and --color-warning-text CSS variables to only apply inside .gt-mono (monospace) contexts. The .attention-tip and .attention-warning blockquotes are outside that scope so they fall back to the body text color #181c21.

Unaffected types: Note (blue), Important (purple), Caution (red) — their variables were not scoped by PR #37019.

/**
 * Regression test for Bug #37416: "Missing colors in Note and Warning markdown highlighting after upgrading to 1.26"
 *
 * Root-Cause: web_src/css/features/colorpicker.css (PR #37019) — Update message severity colors, fix navbar double border
 *   PR #37019 scoped `--color-success-text` (used by Tip alerts) and `--color-warning-text` (used by Warning alerts)
 *   to only apply inside `.gt-mono` (monospace) contexts. The `.attention-tip` and `.attention-warning` blockquote
 *   elements are not inside `.gt-mono`, so they lose their colored CSS variables and fall back to the body text color
 *   `rgb(24, 28, 33)` — making Tip and Warning alerts appear monochrome/black.
 *   Note (info), Important (purple), and Caution (red) are unaffected because they use different CSS variables.
 *
 * v1.26.1 (Bug):  .attention-tip and .attention-warning have borderLeftColor = rgb(24, 28, 33) → MONOCHROME → Test FAIL
 * v1.26.x (Fix):  .attention-tip has a distinct green border; .attention-warning has a distinct yellow/orange border → Test PASS
 */

const { test, expect } = require('@playwright/test');
const fs = require('fs').promises;
const path = require('path');

const RESULTS_DIR = process.env.RESULTS_DIR || path.join(__dirname, 'test-results');

const MARKDOWN_CONTENT = `# Markdown Alert Colors Test

> [!NOTE]
> This is a **note** alert. Border and title should be in blue/info color.

> [!TIP]
> This is a **tip** alert. Border and title should be in green color.

> [!IMPORTANT]
> This is an **important** alert. Border and title should be in purple color.

> [!WARNING]
> This is a **warning** alert. Border and title should be in yellow/orange color.

> [!CAUTION]
> This is a **caution** alert. Border and title should be in red/danger color.
`;

async function screenshot(page, step, label, options = {}) {
  await fs.mkdir(RESULTS_DIR, { recursive: true });
  const filename = `${String(step).padStart(2, '0')}-${label}.png`;
  await page.screenshot({
    path: path.join(RESULTS_DIR, filename),
    fullPage: options.fullPage ?? false,
  });
  return filename;
}

async function apiCall(method, endpoint, body, token) {
  const res = await fetch(`${BASE_URL}${endpoint}`, {
    method,
    headers: {
      'Authorization': `token ${token}`,
      'Content-Type': 'application/json',
    },
    body: body ? JSON.stringify(body) : undefined,
  });
  const text = await res.text();
  let json = null;
  try { json = JSON.parse(text); } catch {}
  return { status: res.status, body: json };
}

  test.beforeAll(async () => {

      method: 'POST',

        'Content-Type': 'application/json',

    // Create repo (or delete+recreate for clean state)

    if (existing.status === 200) {

      name: REPO_NAME,
      description: 'Markdown alert colors regression test #37416',
      private: false,
      auto_init: true,
      default_branch: 'main',
    }, token);
    if (repoRes.status !== 201) throw new Error(`Failed to create repo: ${JSON.stringify(repoRes.body)}`);

    // Push README with alert blocks
    const content = Buffer.from(MARKDOWN_CONTENT).toString('base64');
    // Get SHA of auto-init README
    const fileRes = await apiCall('GET', `/api/v1/repos/${ADMIN_USER}/${REPO_NAME}/contents/README.md`, null, token);
    const sha = fileRes.body?.sha;
    const putRes = await apiCall('PUT', `/api/v1/repos/${ADMIN_USER}/${REPO_NAME}/contents/README.md`, {
      message: 'Add markdown alert blocks for color regression test',

      sha,
    }, token);
    if (putRes.status !== 200 && putRes.status !== 201) {
      throw new Error(`Failed to update README: status=${putRes.status} body=${JSON.stringify(putRes.body)}`);

  test('Tip and Warning alert blocks must have distinct (non-monochrome) border colors', async ({ page }) => {
    // =========================================================================
    // STEP 1: Navigate to repo home page (README renders inline)

    await page.goto(`${BASE_URL}/${ADMIN_USER}/${REPO_NAME}`);
    await page.waitForLoadState('networkidle');
    await screenshot(page, 1, 'repo-home-markdown-alerts', { fullPage: true });
    // BUG (v1.26.1): All five alert blocks visible but Tip and Warning appear in same dark
    //                monochrome color as body text — no colored border/icon/title.
    // FIX: Tip block has a green-tinted border+icon+title; Warning block has yellow/orange.

    // =========================================================================
    // STEP 2: Collect border colors and CSS variable values for all 5 alert types
    // =========================================================================
    const alertColors = await page.evaluate(() => {
      const types = ['note', 'tip', 'important', 'warning', 'caution'];
      const result = {};
      const root = window.getComputedStyle(document.documentElement);
      const bodyTextColor = window.getComputedStyle(document.body).color;

      types.forEach(t => {
        const el = document.querySelector(`.attention-${t}`);
        if (!el) {
          result[t] = { found: false };
          return;
        }
        const cs = window.getComputedStyle(el);
        const borderColor = cs.borderInlineStartColor || cs.borderLeftColor;
        result[t] = {
          found: true,
          borderColor,
          isMonochrome: borderColor === bodyTextColor,
        };
      });

      result.bodyTextColor = bodyTextColor;
      result.cssVarWarningText = root.getPropertyValue('--color-warning-text').trim();
      result.cssVarSuccessText = root.getPropertyValue('--color-success-text').trim();

      return result;
    });

    console.log('\n--- Markdown Alert Border Colors (v1.26.1) ---');
    console.log(`Body text color (monochrome baseline): ${alertColors.bodyTextColor}`);
    ['note', 'tip', 'important', 'warning', 'caution'].forEach(t => {

      }

    console.log(`  --color-warning-text: "${alertColors.cssVarWarningText}" (should be a color hex, not body text)`);

    // FIX: Tip block shows a distinct green border, icon, and title.

    // STEP 6: Assertion — Tip and Warning must NOT use the body text color
